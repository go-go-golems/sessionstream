    81	// and a sessionstream.UIFanout: clients may subscribe/unsubscribe to sessions,
    82	// receive snapshots, and receive live UI events. Command ingress is deliberately
    83	// out of scope for this adapter.

    85	// Subscribe always sends a current snapshot followed by future live UI fanout.

   180	// ServeHTTP upgrades a connection and serves the websocket protocol.
   181	func (s *Server) ServeHTTP(w http.ResponseWriter, r *http.Request) {
   182		conn, err := s.upgrader.Upgrade(w, r, nil)
   183		if err != nil {
   184			s.observe(r.Context(), TransportRecord{Stage: TransportStageUpgradeError, Err: err})
   185			if s.hooks.OnUpgradeError != nil {
   186				s.hooks.OnUpgradeError(r, err)
   187			}
   188			return
   189		}
   190		cid := sessionstream.ConnectionId(fmt.Sprintf("conn-%d", atomic.AddUint64(&s.nextID, 1)))
   191		c := &connection{
   192			id:   cid,
   193			ws:   conn,
   194			send: make(chan outboundFrame, 128),
   195			subs: map[sessionstream.SessionId]subscription{},
   196		}
   197		s.mu.Lock()
   198		s.conns[cid] = c
   199		s.mu.Unlock()
   200		s.observe(r.Context(), TransportRecord{Stage: TransportStageConnected, ConnectionId: cid})
   201		if s.hooks.OnConnect != nil {
   202			s.hooks.OnConnect(cid)
   203		}
   204	
   205		go s.writeLoop(r.Context(), c)
   206		_ = s.sendFrame(c, newHelloFrame(cid))
   207		s.readLoop(r.Context(), c)
   208		s.closeConnection(c)
   209	}

   140			return fmt.Errorf("hub is nil")
   141		}
   142		if p.hub.bus == nil {
   143			return fmt.Errorf("event bus is not configured")
   144		}

   146			return fmt.Errorf("event %q missing session id", ev.Name)
   147		}

   149			return fmt.Errorf("event %q payload is nil", ev.Name)

   151		if err := p.hub.validatePayloadType(p.hub.reg.events, "event", ev.Name, ev.Payload); err != nil {
   152			return err
   153		}
   154	
   155		payload, err := p.hub.reg.MarshalProtoJSON(ev.Payload)
   156		if err != nil {
   157			return fmt.Errorf("marshal event %q payload: %w", ev.Name, err)
   158		}
   159		body, err := json.Marshal(eventEnvelope{
   160			Name:      ev.Name,
   161			SessionID: string(ev.SessionId),
   162			Payload:   payload,
   163		})
   164		if err != nil {
   165			return fmt.Errorf("marshal event %q envelope: %w", ev.Name, err)
   166		}
   167	
   168		msg := message.NewMessage(uuid.NewString(), body)
   169		msg.SetContext(ctx)
   170		msg.Metadata.Set(MetadataKeyEventName, ev.Name)
   171		msg.Metadata.Set(MetadataKeySessionID, string(ev.SessionId))
   172		msg.Metadata.Set(MetadataKeyPartitionKey, PartitionKeyForSession(ev.SessionId))
   173		msg.Metadata.Set(MetadataKeyPublishedOrd, "0")
   174		if p.hub.bus.messageMutator != nil {
   175			if err := p.hub.bus.messageMutator(ctx, Event{Name: ev.Name, SessionId: ev.SessionId, Payload: proto.Clone(ev.Payload)}, msg); err != nil {
   176				return fmt.Errorf("mutate event %q message metadata: %w", ev.Name, err)
   177			}
   178		}
   179		if err := p.hub.bus.publisher.Publish(p.hub.bus.topic, msg); err != nil {
   180			return fmt.Errorf("publish event %q: %w", ev.Name, err)
   181		}
   182		if p.hub.bus.observer != nil {
   183			p.hub.bus.observer.Published(ctx, Event{Name: ev.Name, SessionId: ev.SessionId, Payload: proto.Clone(ev.Payload)}, newBusRecord(msg, p.hub.bus.topic))

   186	}

   188	func decodeEventEnvelope(reg *SchemaRegistry, payload []byte) (Event, error) {
   189		var env eventEnvelope
   190		if err := json.Unmarshal(payload, &env); err != nil {
   191			return Event{}, fmt.Errorf("decode event envelope: %w", err)
   192		}
   193		if env.Name == "" {
   194			return Event{}, fmt.Errorf("event envelope missing name")
   195		}
   196		if env.SessionID == "" {
   197			return Event{}, fmt.Errorf("event envelope missing session id")
   198		}
   199		msg, err := reg.instantiate(reg.events, "event", env.Name)
   200		if err != nil {
   201			return Event{}, err
   202		}
   203		if len(env.Payload) > 0 && string(env.Payload) != "null" {
   204			if err := protojson.Unmarshal(env.Payload, msg); err != nil {
   205				return Event{}, fmt.Errorf("decode event %q payload: %w", env.Name, err)
   206			}
   207		}
   208		return Event{Name: env.Name, SessionId: SessionId(env.SessionID), Payload: msg}, nil
   209	}

   177	func (e *Engine) handleStartInference(ctx context.Context, cmd sessionstream.Command, _ *sessionstream.Session, pub sessionstream.EventPublisher) error {

   190			"messageId": userMessageID,
   191			"role":      "user",
   192			"content":   prompt,
   193			"streaming": false,
   194		}); err != nil {
   195			return err
   196		}
   197		// The demo stream should continue after the command handler returns; cancellation is
   198		// controlled by StopInferenceCommand through the activeRun cancel function. Keep
   199		// context values from the command context, but detach from its cancellation.
   200		runCtx, cancel := context.WithCancel(context.WithoutCancel(ctx))
   201		run := &activeRun{messageID: messageID, cancel: cancel, done: make(chan struct{})}
   202		if previous := e.swapRun(cmd.SessionId, run); previous != nil {
   203			previous.cancel()
   204			<-previous.done
   205		}
   206		// #nosec G118 -- the demo inference goroutine is intentionally detached from
   207		// the command handler lifetime; StopInferenceCommand owns cancellation.
   208		go e.runDemoInference(runCtx, cmd.SessionId, messageID, prompt, pub, run.done)

    33			writeJSON(w, http.StatusMethodNotAllowed, map[string]any{"error": "method not allowed"})
    34			return
    35		}
    36		sessionID := req.URL.Query().Get("sessionId")
    37		if sessionID == "" {
    38			writeJSON(w, http.StatusBadRequest, map[string]any{"error": "sessionId is required"})
    39			return
    40		}
    41		limit := 25
    42		if raw := req.URL.Query().Get("limit"); raw != "" {
    43			parsed, err := strconv.Atoi(raw)
    44			if err != nil || parsed < 0 {
    45				writeJSON(w, http.StatusBadRequest, map[string]any{"error": "limit must be a non-negative integer"})
    46				return
    47			}
    48			limit = parsed
    49		}
    50		resp, err := s.env.inspectPhase5Replay(req.Context(), sessionstream.SessionId(sessionID), limit)
    51		if err != nil {
    52			writeJSON(w, http.StatusInternalServerError, map[string]any{"error": err.Error()})
    53			return
    54		}
    55		writeJSON(w, http.StatusOK, resp)
    56	}
    57	
    58	func (e *labEnvironment) inspectPhase5Replay(ctx context.Context, sid sessionstream.SessionId, limit int) (replayInspectResponse, error) {
    59		e.mu.Lock()
    60		state := e.phase5
    61		e.mu.Unlock()
    62		if state == nil || state.runtime == nil || state.runtime.hub == nil {
    63			return replayInspectResponse{}, fmt.Errorf("phase 5 runtime is not initialized")
    64		}
    65		hub := state.runtime.hub
    66		eventCursor, err := hub.EventCursor(ctx, sid)
    67		if err != nil {
    68			return replayInspectResponse{}, err
    69		}
    70		timelineCursor, err := hub.ProjectionCursor(ctx, sessionstream.TimelineProjectorName, sid)
    71		if err != nil {
    72			return replayInspectResponse{}, err
    73		}
    74		var errorsOut []replayErrorRecord
    75		if errorStore, ok := state.runtime.store.(sessionstream.ErrorRecordStore); ok {
    76			records, err := errorStore.ErrorRecords(ctx, sid, limit)
    77			if err != nil {
    78				return replayInspectResponse{}, err
    79			}
    80			errorsOut = encodeReplayErrors(records)
    81		}
    82		return replayInspectResponse{
    83			Phase:          "phase5",
    84			SessionID:      string(sid),
    85			EventCursor:    eventCursor,
    86			TimelineCursor: timelineCursor,
    87			Errors:         errorsOut,
    88		}, nil
    89	}
    90	
    91	func encodeReplayErrors(records []sessionstream.ErrorRecord) []replayErrorRecord {
    92		if len(records) == 0 {
    93			return []replayErrorRecord{}
    94		}
    95		out := make([]replayErrorRecord, 0, len(records))
    96		for _, rec := range records {
    97			errText := ""
    98			if rec.Err != nil {
    99				errText = rec.Err.Error()
   100			}
   101			raw := ""
   102			if len(rec.RawMessage) > 0 {
   103				raw = base64.StdEncoding.EncodeToString(rec.RawMessage)
   104			}
   105			out = append(out, replayErrorRecord{
   106				Kind:             string(rec.Kind),
   107				SessionID:        string(rec.SessionId),
   108				Ordinal:          rec.Ordinal,
   109				EventName:        rec.EventName,
   110				Error:            errText,
   111				RawMessageBase64: raw,
   112				Metadata:         cloneReplayStringMap(rec.Metadata),
   113			})
   114		}
   115		return out
   116	}
   117	
   118	func cloneReplayStringMap(in map[string]string) map[string]string {
   119		if in == nil {
   120			return nil
   121		}
   122		out := make(map[string]string, len(in))
   123		for k, v := range in {
   124			out[k] = v
   125		}
   126		return out
   127	}
